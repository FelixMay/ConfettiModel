8 * Description:
9 * This file contains function prototypes and class declarations for the C++
10 * library of non-uniform random number generators. Most functions are fast and
11 * accurate, even for extreme values of the parameters.
12 *
13 *
14 * functions without classes:
15 * ==========================
16 *
17 * void EndOfProgram(void);
18 * System-specific exit code. You may modify this to make it fit your
19 * user interface.
20 *
21 * void FatalError(const char * ErrorText);
22 * Used for outputting error messages from the other functions and classes.
23 * You may have to modify this function to make it fit your user interface.
24 *
25 * double Erf (double x);
26 * Calculates the error function, which is the integral of the normal distribution.
27 *
28 * double LnFac(int32_t n);
29 * Calculates the natural logarithm of the factorial of n.
30 *
31 *
32 * class StochasticLib1:
33 * ====================
34 * This class can be derived from any of the uniform random number generators
35 * defined in randomc.h. StochasticLib1 provides the following non-uniform random
36 * variate generators:
37 *
38 * int Bernoulli(double p);
39 * Bernoulli distribution. Gives 0 or 1 with probability 1-p and p.
40 *
41 * double Normal(double m, double s);
42 * Normal distribution with mean m and standard deviation s.
43 *
44 * double NormalTrunc(double m, double s, double limit);
45 * Truncated normal distribution with tails cut off at m +/- limit
46 *
47 * int32_t Poisson (double L);
48 * Poisson distribution with mean L.
49 *
50 * int32_t Binomial (int32_t n, double p);
51 * Binomial distribution. n trials with probability p.
52 *
53 * int32_t Hypergeometric (int32_t n, int32_t m, int32_t N);
54 * Hypergeometric distribution. Taking n items out N, m of which are colored.
55 *
56 * void Multinomial (int32_t * destination, double * source, int32_t n, int colors);
57 * void Multinomial (int32_t * destination, int32_t * source, int32_t n, int colors);
58 * Multivariate binomial distribution.
59 *
60 * void MultiHypergeometric (int32_t * destination, int32_t * source, int32_t n, int colors);
61 * Multivariate hypergeometric distribution.
62 *
63 * void Shuffle(int * list, int min, int n);
64 * Shuffle a list of integers.
65 *
66 *
67 * class StochasticLib2:
68 * =====================
69 * This class is derived from class StochasticLib1. It redefines the functions
70 * Poisson, Binomial and HyperGeometric.
71 * In StochasticLib1, these functions are optimized for being called with
72 * parameters that vary. In StochasticLib2, the same functions are optimized
73 * for being called repeatedly with the same parameters. If your parameters
74 * seldom vary, then StochasticLib2 is faster. The two classes use different
75 * calculation methods, both of which are accurate.
76 *
77 *
78 * class StochasticLib3:
79 * =====================
80 * This class can be derived from either StochasticLib1 or StochasticLib2,
81 * whichever is preferred. It contains functions for generating variates with
82 * the univariate and multivariate Wallenius' and Fisher's noncentral
83 * hypergeometric distributions.
84 *
85 * int32_t WalleniusNCHyp (int32_t n, int32_t m, int32_t N, double odds);
86 * Sampling from Wallenius' noncentral hypergeometric distribution, which is
87 * what you get when taking n items out N, m of which are colored, without
88 * replacement, with bias.
89 *
90 * int32_t FishersNCHyp (int32_t n, int32_t m, int32_t N, double odds);
91 * Sampling from Fisher's noncentral hypergeometric distribution which is the
92 * conditional distribution of independent binomial variates given their sum n.
93 *
94 * void MultiWalleniusNCHyp (int32_t * destination, int32_t * source, double * weights, int32_t n, int colors);
95 * Sampling from multivariate Wallenius' noncentral hypergeometric distribution.
96 *
97 * void MultiFishersNCHyp (int32_t * destination, int32_t * source, double * weights, int32_t n, int colors);
98 * Sampling from multivariate Fisher's noncentral hypergeometric distribution.
99 *
100 *
101 * Uniform random number generators (integer and float) are also available, as
102 * these are inherited from the random number generator class that is the base
103 * class of StochasticLib1.
104 *
105 *
106 * class CWalleniusNCHypergeometric
107 * ================================
108 * This class implements various methods for calculating the probability
109 * function and the mean and variance of the univariate Wallenius' noncentral
110 * hypergeometric distribution. It is used by StochasticLib3 and can also be
111 * used independently.
112 *
113 *
114 * class CMultiWalleniusNCHypergeometric
115 * =====================================
116 * This class implements various methods for calculating the probability func-
117 * tion and the mean of the multivariate Wallenius' noncentral hypergeometric
118 * distribution. It is used by StochasticLib3 and can also be used independently.
119 *
120 *
121 * class CMultiWalleniusNCHypergeometricMoments
122 * ============================================
123 * This class calculates the exact mean and variance of the multivariate
124 * Wallenius' noncentral hypergeometric probability distribution.
125 *
126 *
127 * class CFishersNCHypergeometric
128 * ==============================
129 * This class calculates the probability function and the mean and variance
130 * of Fisher's noncentral hypergeometric distribution.
131 *
132 *
133 * class CMultiFishersNCHypergeometric
134 * ===================================
135 * This class calculates the probability function and the mean and variance
136 * of the multivariate Fisher's noncentral hypergeometric distribution.
137 *

253 // The following tables are tables of residues of a certain expansion
254 // of the error function. These tables are used in the Laplace method
255 // for calculating Wallenius' noncentral hypergeometric distribution.
256 // There are ERFRES_N tables covering desired precisions from
257 // 2^(-ERFRES_B) to 2^(-ERFRES_E). Only the table that matches the
258 // desired precision is used. The tables are defined in erfres.h which
259 // is included in wnchyppr.cpp.
260 
261 // constants for ErfRes tables:
262 static const int ERFRES_B = 16; // begin: -log2 of lowest precision
263 static const int ERFRES_E = 40; // end: -log2 of highest precision
264 static const int ERFRES_S = 2; // step size from begin to end
265 static const int ERFRES_N = (ERFRES_E-ERFRES_B)/ERFRES_S+1; // number of tables
266 static const int ERFRES_L = 48; // length of each table
267 
268 // tables of error function residues:
269 extern "C" double ErfRes [ERFRES_N][ERFRES_L];
270 
271 // number of std. deviations to include in integral to obtain desired precision:
272 extern "C" double NumSDev[ERFRES_N];
